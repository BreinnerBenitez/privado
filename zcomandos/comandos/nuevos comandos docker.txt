docker pull node = descargar imagen para contenedor
docker image rm node:18 = borra la imagen
docker rm + nombre = borra el contenedor ejcutado apartir de la imagen
docker create mongo o docker container create mongo (apartir de  este contenedor crea uno) = crear contenedor
docker logs + id = mosrar los log para ver que paso
docker build -t nombre_imagen (miapp:1) . = para crear imagen docker a partir del docker file
docker ps -a = mostrar contenedores detenidos 
docker exec -it mysql-container = puedo ingresar al contenedor docker con terminal
docker create -p27017:27017 --name nombre_contenedor nombreimagen = crea contenedor con puerto a partir de la imagen
docker network ls = listar todas las redes docker
docker network create mired = crear red 
docker network rm mired = borra la red
docker network connect mi_red miapp = conectar un contenedor a una red
--network mired = esta opcion  agrupa el contenedor a la red
docker network disconnect mired mysql-container = desconecta contenedor de la red
docker network inspect mired = inspecionar la red
